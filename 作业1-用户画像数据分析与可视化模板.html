# 1. 统计每个用户在每个年月、每个商品类别的购买次数
user_month_cate = consume_df.groupby(['消费年月', '商品类别', '用户ID']).size().reset_index(name='购买次数')
#groupby(['消费年月', '商品类别', '用户ID'])：这是最关键的“分类堆放”指令。

#把原始的消费记录表拿过来，按照这三个条件分成一堆一堆的。
#比如：【2024年5月 + 机票 + 张三】放在一堆，【2024年5月 + 机票 + 李四】放在另一堆。
#.size()：数一数【2024年5月 + 机票 + 张三】这一堆里有几张小票？
#如果是3张，说明张三在这个月买了3次机票。
#.reset_index(name='购买次数')：把数出来的结果做成一张漂亮的新表格，
# 给数量那一列起名叫“购买次数”，并把整张新表存进 user_month_cate 这个新抽屉里。


# 2. 统计各年月各品类的总购买人数（活跃买家数）
total_buyers = user_month_cate.groupby(['消费年月', '商品类别']).size().reset_index(name='总人数')
#把刚才第一步做好的那张表拿过来，把【2024年5月 + 机票】的所有人都归为一堆
# （不管张三、李四还是王五，只要是这个月买机票的，都在这一堆里）。
#.size()：再数数这一堆里有多少行。因为第一步里，每个人已经凝聚成了一行（比如张三只占一行），
# 所以这时候数出来的行数，恰好就是这个月买机票的总人数（不重复的总人数）。
#name='总人数'：把结果起名叫“总人数”，存入 total_buyers 抽屉。


# 3. 统计各年月各品类的复购人数（购买次数 >= 2）
repurchase_buyers = user_month_cate[user_month_cate['购买次数'] >= 2].groupby(['消费年月', '商品类别']).size().reset_index(name='复购人数')
#user_month_cate['购买次数'] >= 2：【条件筛选】 机器人先拿把筛子，把第一步那张表里，购买次数小于 2（也就是只买过 1 次）的人全部筛掉。
# 只留下那些买了 2 次、3 次、甚至更多次的老顾客。
#.groupby(['消费年月', '商品类别'])：【再次分组】 把这些留下来的老顾客，
# 按照【月份】和【品类】堆在一起。比如【2024年5月 + 机票】堆一堆。
#.size() / name='复购人数'：【数数命名】 数一数这一堆里有多少个老顾客，起名叫“复购人数”，存入 repurchase_buyers 抽屉。


# 4. 合并数据并计算复购率
repurchase_df = pd.merge(total_buyers, repurchase_buyers, on=['消费年月', '商品类别'], how='left').fillna(0)
#pd.merge(total_buyers, repurchase_buyers, ...)：把“总人数表”和“复购人数表”像 Excel 里的 VLOOKUP 一样拼起来。
#on=['消费年月', '商品类别']：按照月份和商品品类对齐。比如把 202405 的机票总人数，和 202405 的机票复购人数并排拼在一行。
#how='left'：以左边的“总人数表”为主。如果某个冷门品类当月有人买，但没有任何人复购，右边表里就会没有记录，就会在表格里留一个表示空值的 NaN。
#.fillna(0)：【打补丁】它把所有拼表产生的 NaN（空值）全部填上数字 0。意思是：没人复购，那复购人数就是 0 人。

repurchase_df['复购率'] = repurchase_df['复购人数'] / repurchase_df['总人数']
#做除法：在表格最右边强行新开一列叫“复购率”。
#计算公式：复购率 = 复购人数 ÷ 总人数。比如总共 100 个人买机票，有 20 个人买了 2 次以上，那复购率就是 20%。

#优惠卷投放情况（数据分析及可视化代码）

# 统计各品类下是否使用优惠券的订单量

coupon_analysis = consume_df.groupby(['商品类别', '是否使用优惠券']).size().unstack(fill_value=0)
#groupby(['商品类别', '是否使用优惠券'])：把消费表拿过来，先按“商品类别（机票/酒店等）”堆成一堆，再在里面按“是否使用优惠券（是/否）”分成更小的堆。
#.size()：数数每个小堆里有多少张订单。
#.unstack(fill_value=0)：unstack 叫“行转列（透视）”。
#默认算完的数据是长条形的，很难看。加上这句后，把数据扭转成一个像 Excel 交叉表一样的方块表。横行是“商品类别”，纵列是“是”和“否”。
#fill_value=0：如果某个品类（比如‘当地玩乐’）没有任何一单用过券，格子原本会是空的（NaN），这里指定用 0 填满。
coupon_analysis['优惠券覆盖率'] = coupon_analysis['是'] / (coupon_analysis['是'] + coupon_analysis['否'])
#算比例：在表格最右边新开一列，起名叫“优惠券覆盖率”。
#计算公式：用券单数（是） ÷ 总单数（是 + 否）。这就帮运营算出了每个品类的领券转化效果。
print("--- 优惠券各品类覆盖率统计 ---")
print(coupon_analysis)

# 1. 计算每个用户的累计消费总额
user_spend = consume_df.groupby('用户ID')['消费金额(元)'].sum().reset_index()
#groupby('用户ID')：机器人来到消费记录表，把所有小票按用户堆成一堆。张三的小票一堆，李四的一堆。
#['消费金额(元)'].sum()：sum 是求和。机器人把张三这堆小票上的“消费金额”全部加起来，算出张三总共在平台花了多少钱。
#.reset_index()：重新整理成一张干净的“会员账单表”，起名叫 user_spend。
#表格里长这样：【张三，累计消费5000元】、【李四，累计消费200元】。

# 2. 划定 Top 20% 高消费用户的金额阈值
threshold = user_spend['消费金额(元)'].quantile(0.80)
#.quantile(0.80)：分位数函数，这是找财富分界线的黑科技。
#把所有人按照累计消费金额，从低到高排成一条长队。
#quantile(0.80) 意思是：走到队伍 80% 的位置，看看那个人的钱包里有多少钱。这个金额就是 threshold（门槛线/阈值）。
#比如门槛是 3000 元，意味着只有消费超过 3000 元的人，才能排进前 20% 的富豪榜。
top_20_users = user_spend[user_spend['消费金额(元)'] >= threshold]
#用这根门槛线当筛子，把所有消费大于等于门槛金额的土豪挑出来，存进 top_20_users（高价值用户表）中。
print(f"Top 20% 高消费用户入围门槛为累计消费：{threshold:.2f}元，高价值用户共：{len(top_20_users)}人")

# 3. 与用户行为偏好表进行 Inner Merge 联动
top_20_profile = pd.merge(top_20_users, behavior_df, on='用户ID', how='inner')
#把“土豪表”和“行为表”并排拼在一起，按照“用户ID”对齐。
#how='inner'（内连接）就像是一把严格的锁：必须这个人既在土豪表里（是VIP），又在行为表里（有干净的行为记录），才保留下来。
#拼完后的 top_20_profile 表里，就同时拥有了这群 VIP 的消费金额、性别、年龄、以及他们最爱看的“内容偏好”。

# 4. 统计高价值用户的内容偏好分布比例
top_20_content = top_20_profile.groupby('内容偏好').size().reset_index(name='高价值用户数')
top_20_content['占比'] = top_20_content['高价值用户数'] / top_20_content['高价值用户数'].sum()
#新开一列叫“占比”。公式是：该偏好的土豪人数 ÷ 土豪总人数。
print("\n--- Top 20% 高消费用户的内容偏好分布 ---")
print(top_20_content)

# 5. 可视化高价值用户偏好
plt.figure(figsize=(8, 5))
sns.barplot(data=top_20_content.sort_values(by='占比', ascending=False), x='内容偏好', y='占比', palette='copper')
plt.title('TravelGo 平台高消费用户（Top 20%）首页内容偏好分布图', fontsize=14)
plt.ylabel('高价值用户群体内部占比')
plt.gca().yaxis.set_major_formatter(plt.FuncFormatter(lambda y, _: '{:.0%}'.format(y)))
#这一行是高端格式化黑科技！ * 默认情况下，纵坐标轴显示的数字是小数（比如 0.1、0.2、0.3），在 PPT 汇报里看起来不够直观。
#这行代码是强行命令：“把纵坐标轴的数字，全部自动乘以 100 并且加上百分号（%），也就是把 0.45 变成 45% 显示出来！”
plt.tight_layout()
plt.show()
